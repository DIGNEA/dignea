163 template <class S>
164 ExpBuilder<S>::operator unique_ptr<Experiment<S>>() {
165  if (!this->evolutionaryAlgorithm) {
166  throw std::runtime_error(
167  "EA not set in ExpBuilder. Use usingAlgorithm()");
168  }
169  if (!this->problem) {
170  throw std::runtime_error(
171  "Problem not set in ExpBuilder. Use toSolve()");
172  }
173  if (outputPath.empty()) {
174  throw std::runtime_error(
175  "Output path not set in ExpBuilder. Use saveResultsIn()");
176  }
177 
178  if (repetitions == 0) {
179  throw std::runtime_error(
180  "Repetitions equal to 0 in ExpBuilder. Use repeat()");
181  }
182  string finalName = this->outputPath + "/" + this->experimentName;
183  auto experiment = make_unique<Experiment<S>>(
184  move(problem), move(evolutionaryAlgorithm), finalName, repetitions);
185  return experiment;
186 }
