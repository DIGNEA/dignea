79 template <typename MS>
80 vector<vars> NSFeatures<MS>::beforeRun(vector<MS> &population) {
81  vector<vars> combinedPop;
82  combinedPop.reserve(population.size() + this->noveltyArchive.size());
83  for (MS &solution : population) {
84  combinedPop.push_back(solution.getFeatures());
85  }
86  combinedPop.insert(combinedPop.end(), this->noveltyFeatures.begin(),
87  this->noveltyFeatures.end());
88  return combinedPop;
89 }

102 template <typename MS>
103 vector<MS> NSFeatures<MS>::run(vector<MS> &population,
104  const Problem<MS> *problem) {
105  vector combinedPop = this->beforeRun(population);
106  vector<float> spars =
107  sparseness(combinedPop, this->distance.get(), this->k);
108  for (unsigned int i = 0; i < population.size(); i++) {
109  population[i].setDiversity(spars[i]);
110  }
111  return population;
