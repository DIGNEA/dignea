41 template <class S>
42 unique_ptr<AbstractGA<S>> createDefaultGA(GAType type) {
43  map<GAType, function<unique_ptr<AbstractGA<S>>()>> factories;
44  factories[GAType::FIGA] = [] { return make_unique<FIGA<S>>(); };
45  factories[GAType::Generational] = [] {
46  return make_unique<GenerationalGA<S>>();
47  };
48  factories[GAType::Steady] = [] { return make_unique<SteadyGA<S>>(); };
49  factories[GAType::Parallel] = [] {
50  return make_unique<ParallelGeneticAlgorithm<S>>();
51  };
52 
53  return factories[type]();
54 }

291 template <class S>
292 GABuilder<S>::operator unique_ptr<AbstractGA<S>>() {
293  if (!this->problemSet) {
294  std::cerr << "Algorithm created but no problem set. "
295  "Remember to set a problem before calling run"
296  << std::endl;
297  }
298  if (compSet >= MUST_COMPS)
299  return move(this->ga);
300  else {
301  throw runtime_error(
302  "No all the components are set. GA requires: problem, "
303  "mutation operator, crossover operator, selection operator, "
304  "mutation rate, crossover rate, population size "
305  "and max evaluations");
306  }
