288 template <typename IP, typename IS, typename OP, typename OS>
289 void EIG<IP, IS, OP, OS>::run() {
290  if (instProb) {
291 #ifdef DEBUG
292  std::cout << "Running EIG" << std::endl;
293 #endif
294  createInitialPopulation();
295  this->startTime = chrono::system_clock::now();
296  vector<IS> offspring(this->populationSize);
297  while (!isStoppingConditionReached()) {
298 #ifdef DEBUG
299  std::cout << "Performing Generation " << this->performedEvaluations
300  << std::endl;
301 #endif
302  for (int i = 0; i < this->populationSize; i++) {
303  IS firstChild = selection->select(this->population);
304  IS secondChild = selection->select(this->population);
305  reproduction(firstChild, secondChild);
306  offspring[i] = firstChild;
307  }
308  evaluationPhase(offspring);
309  replacement(offspring);
310  // Updates the fitness evolution
311  this->updateEvolution(this->population);
312  updateProgress();
313  }
314  this->endTime = chrono::system_clock::now();
315  finishProgress();
316  } else {
317  throw std::logic_error("AbstractDomain not set in EIG::run");
318  }
319 }

459 template <typename IP, typename IS, typename OP, typename OS>
460 void EIG<IP, IS, OP, OS>::evaluatePopulation(vector<IS> &individuals) {
461  instProb->beforeEvaluation(individuals);
462  for (IS &individual : individuals) {
463  // Creates the optimisation problem for the EAs
464  shared_ptr<OP> optProblem = instProb->genOptProblem(individual);
465 
466  vector<float> avgBestFitEAs; // Avg. Best fitness from each alg.
467  avgBestFitEAs.reserve(algPortfolio.size());
468  vector<vector<float>> bestSolsEAs; // Best #reps solutions/algorithm
469 
470  for (unique_ptr<Alg> &config : algPortfolio) {
471  vector<float> bestF(repetitions, 0.0); // Resets the bestF
472  // Run the problem rep times
473  for (int rep = 0; rep < repetitions; rep++) {
474  config->setProblem(optProblem);
475  config->run();
476  // Get the results and compute the best founds
477  Front<OS> front = config->getResults();
478  OS bestSolution = bestFitness.getBestSolution(front);
479  bestF[rep] = bestSolution.getFitness();
480  }
481  // Saves avg. of the best found in #reps
482  float averageBestFitness = averageFitness.computeValue(bestF);
483  avgBestFitEAs.push_back(averageBestFitness);
484  // Saves all best #reps solutions for analysis
485  bestSolsEAs.push_back(bestF);
486  }
487 
488  // Saves the fitness founds to future analysis
489  individual.setConfigFitness(bestSolsEAs);
490  individual.setAvgConfigFitness(avgBestFitEAs);
491  // Set the biased fitness depending on the type of instance to generate
492  // (easy or hard)
493  individual.setBiasedFitness(
494  this->instanceFitness->compute(avgBestFitEAs));
495  }
496  // Problem dependent changes that must be executed after evaluate the
497  // instances
498  instProb->afterEvaluation(individuals);
499 }

511 template <typename IP, typename IS, typename OP, typename OS>
512 [[maybe_unused]] void EIG<IP, IS, OP, OS>::evaluateIndividual(IS &individual) {
513  // Creates the optimisation problem for the EAs
514  shared_ptr<OP> optProblem = instProb->genOptProblem(individual);
515  vector<float> allFitness;
516  allFitness.reserve(algPortfolio.size());
517  for (unique_ptr<Alg> &config : algPortfolio) {
518  Front<OS> solutions(repetitions);
519  for (int rep = 0; rep < repetitions; rep++) {
520  config->setProblem(optProblem);
521  config->run();
522  Front<OS> front = config->getResults();
523  OS bestSolution = bestFitness.getBestSolution(front);
524  solutions.addSolution(bestSolution);
525  }
526  // Set the fitness as the avg. of the best found in #repetitions times
527  float averageBestFitness = averageFitness.computeValue(solutions);
528  allFitness.push_back(averageBestFitness);
529  }
530  individual.setBiasedFitness(instanceFitness->compute(allFitness));
531 }

562 template <typename IP, typename IS, typename OP, typename OS>
563 void EIG<IP, IS, OP, OS>::replacement(vector<IS> &individuals) {
564  if (individuals.size() != this->population.size()) {
565  std::string where = " replacement at EIG";
566  throw(Mismatch(where));
567  }
568  const float probIncl = this->populationSize / 100.0;
569  for (int i = 0; i < this->populationSize; i++) {
570  if (cmpByFitness(individuals[i], this->population[i]) == FIRST_BEST) {
571  this->population[i] = individuals[i];
572  }
573  if ((PseudoRandom::randDouble() > probIncl) &&
574  (this->population[i].getBiasedFitness() > 0)) {
575  // Always check feasibility
576  this->noveltySearch->insertIntoArchive(this->population[i]);
577  }
578  }
579  // Novelty Search procedures to include individuals in the archives
580  this->noveltySearch->cmpFinals(this->population);
581 }
