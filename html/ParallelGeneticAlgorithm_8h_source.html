178 template <class S>
179 void ParallelGeneticAlgorithm<S>::runEvolution() {
180  const int popDim = this->populationSize;
181  const int MAX_EVALS = this->maxEvaluations;
182  const float eps = 1e-9;
183 #pragma omp parallel default(none) shared(popDim, MAX_EVALS, eps)
184  {
185  int evals = popDim;
186  int numberOfThreads = omp_get_num_threads();
187  int tid = omp_get_thread_num();
188  int myMaxEvals = MAX_EVALS / numberOfThreads;
189  if (tid < MAX_EVALS % numberOfThreads) myMaxEvals += popDim;
190  do {
191 #pragma omp for schedule(dynamic, chunks) nowait
192  for (int i = 0; i < popDim; ++i) {
193  // Splits the population in chunks to perform the selection
194  int init = tid * chunks;
195  int end = init + (chunks - 1);
196  S child1 = parallelSelection(init, end);
197  S child2 = parallelSelection(init, end);
198  this->reproduction(child1, child2);
199  problem->evaluate(child1);
200  // Replacement
201  bool update = false;
202  double childPenalty = child1.getConstraintCoeff();
203  double individualPenalty = individuals[i].getConstraintCoeff();
204  double childFitness = child1.getFitness();
205  double individualFitness = individuals[i].getFitness();
206  // If child has less penalty or in equal penalty values it has
207  // better fitness
208  if (childPenalty < individualPenalty) {
209  update = true;
210  } else if ((abs(childPenalty - individualPenalty) < eps) &&
211  (childFitness > individualFitness)) {
212  update = true;
213  }
214  if (update) {
215  individuals[i] = child1;
216  }
217  }
218  // Updating the performed evaluations
219  evals += this->populationSize;
220  } while (evals < myMaxEvals);
221  }
222  // Updates superclass variables with this variables used for parallel
223  // executions
224  this->population = this->individuals;
225  this->performedEvaluations = this->maxEvaluations;
226 }

252 S ParallelGeneticAlgorithm<S>::parallelSelection(const int &init,
253  const int &end) {
254  int firstOption = prng.randInt(init, end);
255  int secondOption = prng.randInt(init, end);
256  S child = individuals[min(firstOption, secondOption)];
