124 template <typename S, typename T>
125 vector<S> NoveltySearch<S, T>::run(vector<S> &population,
126  const Problem<S> *problem) {
127  vector<S> combinedPop = population;
128  combinedPop.insert(combinedPop.end(), noveltyArchive.begin(),
129  noveltyArchive.end());
130  vector<T> spars = sparseness(combinedPop, distance.get(), k);
131  for (unsigned int i = 0; i < population.size(); i++) {
132  population[i].setDiversity(spars[i]);
133  }
134  return population;
135 }

146 template <typename S, typename T>
147 void NoveltySearch<S, T>::cmpFinals(vector<S> &population) {
148  if (population.size() == 0) {
149  std::string where = "NoveltySearch::cmpFinals";
150  throw EmptyPopulation(where);
151  }
152  // Includes fittest feasible individual if the archive is empty
153  if (finalSs.empty()) {
154  sortByFitness(population);
155  for (S &sol : population) {
156  if (sol.getFitness() > 0) {
157  finalSs.push_back(population[0]);
158  break;
159  }
160  }
161  } else {
162  // Includes feasible individuals with spars > threshold
163  for (S &solution : population) {
164  T spars = sparseness(solution, finalSs, distance.get(), 1);
165  if ((solution.getBiasedFitness() > 0) &&
166  (spars > this->threshold)) {
167  finalSs.push_back(solution);
168  }
169  }
170  }
171 }
