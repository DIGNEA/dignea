144  BestFitness<S> bestFitness;
145  int rep = 0;
146  while (rep < repetitions) {
147  algorithm->setProblem(problemToSolve);
148  algorithm->run();
149  if (!algorithm->getResults().empty()) {
150  Front<S> solutions = algorithm->getResults();
151  repsSolutions.push_back(solutions);
152  bestFounds[rep] = bestFitness.computeValue(solutions);
153  rep++;
154  }
155  }
