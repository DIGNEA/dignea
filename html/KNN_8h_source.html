139 neighMatrix<T> KNN::run(const vector<S> &population, Distance<T> *metric,
140  const int &k) {
141  if ((unsigned int)k > population.size()) {
142  throw runtime_error(
143  "K is " + to_string(k) +
144  " and population are only: " + to_string(population.size()));
145  }
146  neighMatrix<T> nearest;
147  nearest.reserve(population.size());
148  for (unsigned int i = 0; i < population.size(); i++) {
149  neighVector<T> distances = {};
150  distances.reserve(population.size() - 1);
151  vector tempVars = population[i].getVariables();
152  vector<T> iVars(tempVars.begin(), tempVars.end());
153  for (unsigned int j = 0; j < population.size(); j++) {
154  // For each other individual we compute the distance to i
155  if (i == j) continue;
156  vector tmpJVars = population[j].getVariables();
157  vector<T> jVars(tmpJVars.begin(), tmpJVars.end());
158  T dist = metric->compute(iVars, jVars);
159  distances.push_back({dist, j});
160  }
161  // Ordenamos los vecinos de I por su distancia a I
162  sort(distances.begin(), distances.end());
163  // We only get the K nearest neighbors
164  nearest.push_back({distances.begin(), distances.begin() + k});
165  }
166  return nearest;
167 }
