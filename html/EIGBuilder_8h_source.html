125 template <typename IP, typename IS, typename OP, typename S>
126 EIGBuilder<IP, IS, OP, S>::operator unique_ptr<EIG<IP, IS, OP, S>>() {
127  if (!this->problemSet) {
128  std::cerr << "EIG created but not problem set. "
129  "Remember to set a problem before calling run"
130  << std::endl;
131  }
132  if (compSet >= MUST_COMPS) {
133  return move(this->generator);
134  } else {
135  throw runtime_error(
136  "No all the components are set. EIG requires: "
137  "configurations, evaluation method,"
138  "Novelty Search configuration, "
139  "mutation operator, crSsover operator, "
140  "selection operator, mutation rate, "
141  "crSsover rate, population size, "
142  "max generations and repetitions");
143  }
144 }
